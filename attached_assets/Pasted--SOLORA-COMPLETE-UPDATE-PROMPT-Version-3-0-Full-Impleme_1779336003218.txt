### Cursor Movement System

Use lerp (linear interpolation) for smooth cinematic motion:

```javascript
// Cursor tracking with lerp smoothing
let mouse = { x: 0, y: 0 };
let lens = { x: 0, y: 0 };
let ring = { x: 0, y: 0 };

const LENS_LERP = 0.12;   // Inner lens — slightly faster
const RING_LERP = 0.07;   // Outer ring — more lag = more luxury

document.addEventListener('mousemove', (e) => {
  mouse.x = e.clientX;
  mouse.y = e.clientY;
});

function animateCursor() {
  // Lens follows with slight lag
  lens.x += (mouse.x - lens.x) * LENS_LERP;
  lens.y += (mouse.y - lens.y) * LENS_LERP;

  // Ring follows with more lag
  ring.x += (mouse.x - ring.x) * RING_LERP;
  ring.y += (mouse.y - ring.y) * RING_LERP;

  // Apply transforms (center-corrected)
  cursorLens.style.transform = `translate(${lens.x - 24}px, ${lens.y - 24}px)`;
  cursorRing.style.transform = `translate(${ring.x - 36}px, ${ring.y - 36}px)`;
  cursorHandle.style.transform = `translate(${lens.x + 16}px, ${lens.y + 16}px) rotate(45deg)`;

  requestAnimationFrame(animateCursor);
}

animateCursor();
```

The cursor should feel:
**fluid, inertial, alive — never snappy or mechanical.**
